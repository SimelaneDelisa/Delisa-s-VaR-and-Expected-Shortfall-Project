BACKTESTING VAR¶


















In [23]:



     


# Step 1: Define your VaR threshold 
test_confidence_level = 0.95
VaR_threshold = -np.percentile(range_returns, 100 - (test_confidence_level * 100)) * portfolio_value

# Step 2: Calculate actual P&L over the same horizon (already done)
# Use range_returns * portfolio_value to convert to actual returns in Rand
actual_losses = range_returns * portfolio_value

# Step 3: Identify violations (where actual loss is worse than VaR)
violations = actual_losses < -VaR_threshold

# Step 4: Calculate violation rate
total_periods = len(actual_losses)
violation_count = violations.sum()
violation_rate = violation_count / total_periods

print(f"\nBacktesting {int(test_confidence_level*100)}% VaR:")
print(f"VaR threshold: R{VaR_threshold:,.2f}")
print(f"Total periods tested: {total_periods}")
print(f"Violations (actual loss exceeded VaR): {violation_count}")
print(f"Violation rate: {violation_rate*100:.2f}%")
print(f"Expected violation rate: {(1 - test_confidence_level)*100:.2f}%")

# === 1. Set backtest parameters ===
days = 5
portfolio_value = 10000
confidence_level = 0.95
VaR_threshold = -np.percentile(range_returns, 100 - (confidence_level * 100)) * portfolio_value

# === 2. Compute actual 5-day rolling portfolio returns in Rands ===
actual_returns_rands = range_returns * portfolio_value

# === 3. Identify violations where loss exceeds historical VaR ===
violations = actual_returns_rands < -VaR_threshold
total_periods = len(actual_returns_rands)
violation_count = violations.sum()
violation_rate = violation_count / total_periods

# === 4. Print Summary ===
print(f"\nBacktesting Historical VaR at {int(confidence_level * 100)}% Confidence Level")
print(f"Total Periods: {total_periods}")
print(f"Number of Violations (Loss > VaR): {violation_count}")
print(f"Expected Violations: {int((1 - confidence_level) * total_periods)}")
print(f"Violation Rate: {violation_rate:.2%}")

# === 5. Create DataFrame for Plotting ===
backtest_df = pd.DataFrame({
    'Actual_Return': actual_returns_rands,
    'Historical_VaR': -VaR_threshold
}, index=range_returns.index)

# === 6. Plot ===
plt.figure(figsize=(12, 6))
plt.plot(backtest_df.index, backtest_df['Actual_Return'], label='Actual 5-Day Return (R)', alpha=0.6)
plt.plot(backtest_df.index, backtest_df['Historical_VaR'], label=f'{int(confidence_level * 100)}% Historical VaR (R)', color='red')
plt.fill_between(backtest_df.index, backtest_df['Historical_VaR'], backtest_df['Actual_Return'],
                 where=violations, color='orange', label='Violation', alpha=0.4)
plt.legend()
plt.title(f'Backtesting {int(confidence_level * 100)}% Historical VaR over {days}-Day Horizon')
plt.xlabel('Date')
plt.ylabel('Return (Rand)')
plt.grid(True)
plt.tight_layout()
plt.show()
