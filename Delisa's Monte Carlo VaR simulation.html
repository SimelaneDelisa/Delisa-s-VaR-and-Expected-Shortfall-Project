create portfolio expected return¶


















In [8]:



     


def expected_return(weights,log_returns):
    return np.sum(log_returns.mean()*weights)
 



     























create a function used to calculate standard deviation¶


















In [9]:



     


def standard_deviation(weights, cov_matrixx):
    variance = weights.T @ cov_matrixx @ weights
    return np.sqrt(variance)

days = 5

def scenario_gain_loss(portfolio_value, portfolio_std_dev, z_score, days):
    return portfolio_value * portfolio_expected_return * days + portfolio_value * portfolio_std_dev * z_score * np.sqrt(days)

# === 1. Set parameters ===
days = 5
portfolio_value = 10000
confidence_level = 0.95

# === 2. Monte Carlo VaR already computed from earlier simulations ===
# VaR_mc = -np.percentile(scenario_return, 100*(1 - confidence_level))  # Already done

# Convert to monetary value (if not already)
VaR_mc_rands = VaR  # Already in Rands in your Monte Carlo calculation

# === 3. Compute actual 5-day rolling portfolio returns (as before) ===
# Assuming 'historical_returns' is daily log return of the portfolio
range_returns_mc = historical_returns.rolling(days).sum().dropna()
actual_returns_rands_mc = range_returns_mc * portfolio_value

# === 4. Identify violations where actual loss > simulated VaR ===
violations_mc = actual_returns_rands_mc < -VaR_mc_rands
total_periods_mc = len(actual_returns_rands_mc)
violation_count_mc = violations_mc.sum()
violation_rate_mc = violation_count_mc / total_periods_mc

# === 5. Print Summary ===
print(f"\nBacktesting Monte Carlo VaR at {int(confidence_level * 100)}% Confidence Level")
print(f"Total Periods: {total_periods_mc}")
print(f"Number of Violations (Loss > VaR): {violation_count_mc}")
print(f"Expected Violations: {int((1 - confidence_level) * total_periods_mc)}")
print(f"Violation Rate: {violation_rate_mc:.2%}")

# === 6. Create DataFrame for Plotting ===
backtest_df_mc = pd.DataFrame({
    'Actual_Return': actual_returns_rands_mc,
    'MonteCarlo_VaR': -VaR_mc_rands
}, index=range_returns_mc.index)

# === 7. Plot ===
plt.figure(figsize=(12, 6))
plt.plot(backtest_df_mc.index, backtest_df_mc['Actual_Return'], label='Actual 5-Day Return (R)', alpha=0.6)
plt.plot(backtest_df_mc.index, backtest_df_mc['MonteCarlo_VaR'], label=f'{int(confidence_level * 100)}% Monte Carlo VaR (R)', color='red')
plt.fill_between(backtest_df_mc.index, backtest_df_mc['MonteCarlo_VaR'], backtest_df_mc['Actual_Return'],
                 where=violations_mc, color='orange', label='Violation', alpha=0.4)
plt.legend()
plt.title(f'Backtesting {int(confidence_level * 100)}% Monte Carlo VaR over {days}-Day Horizon')
plt.xlabel('Date')
plt.ylabel('Return (Rand)')
plt.grid(True)
plt.tight_layout()
plt.show()
